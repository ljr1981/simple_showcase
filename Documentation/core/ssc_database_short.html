	SSC_DATABASE

create 
	make

feature -- Access

	db: SIMPLE_SQL_DATABASE
			-- Database connection

	db_path: STRING_8
			-- Path to database file
	
feature -- Analytics

	log_request (a_path, a_method, a_ip, a_user_agent, a_referrer: STRING_8; a_response_code, a_response_time_ms: INTEGER_32)
			-- Log an HTTP request to analytics table
		require
			db_open: db.is_open
			path_not_empty: not a_path.is_empty
			method_not_empty: not a_method.is_empty

	get_page_view_counts: ARRAYED_LIST [TUPLE [path: STRING_8; view_count: INTEGER_32]]
			-- Get page view counts grouped by path, ordered by count desc
		ensure
			result_attached: Result /= Void

	get_recent_requests (a_limit: INTEGER_32): ARRAYED_LIST [TUPLE [path: STRING_8; method: STRING_8; ip: STRING_8; created_at: STRING_8]]
			-- Get most recent requests
		require
			limit_positive: a_limit > 0
		ensure
			result_attached: Result /= Void

	get_analytics_summary: TUPLE [total_requests: INTEGER_32; unique_paths: INTEGER_32; unique_ips: INTEGER_32]
			-- Get summary statistics
	
feature -- Contacts

	save_contact (a_name, a_email, a_subject, a_message, a_ip: STRING_8): INTEGER_64
			-- Save contact form submission, return ID
		require
			db_open: db.is_open
			name_not_empty: not a_name.is_empty
			email_not_empty: not a_email.is_empty
			message_not_empty: not a_message.is_empty

	get_contacts (a_limit: INTEGER_32): ARRAYED_LIST [TUPLE [id: INTEGER_64; name: STRING_8; email: STRING_8; subject: STRING_8; message: STRING_8; created_at: STRING_8; read_at: detachable STRING_8]]
			-- Get contact submissions
		require
			limit_positive: a_limit > 0
		ensure
			result_attached: Result /= Void

	mark_contact_read (a_id: INTEGER_64)
			-- Mark contact as read
		require
			db_open: db.is_open
			id_positive: a_id > 0

	get_unread_contact_count: INTEGER_32
			-- Count of unread contacts
	
feature -- Sessions

	create_session (a_token: STRING_8; a_hours: INTEGER_32): BOOLEAN
			-- Create admin session with expiration
		require
			db_open: db.is_open
			token_not_empty: not a_token.is_empty
			hours_positive: a_hours > 0

	validate_session (a_token: STRING_8): BOOLEAN
			-- Check if session token is valid and not expired
		require
			db_open: db.is_open
			token_not_empty: not a_token.is_empty

	delete_session (a_token: STRING_8)
			-- Delete session (logout)
		require
			db_open: db.is_open
			token_not_empty: not a_token.is_empty

	cleanup_expired_sessions
			-- Remove expired sessions
		require
			db_open: db.is_open
	
feature -- Lifecycle

	close
			-- Close database connection
		ensure
			closed: not db.is_open
	
invariant
	db_attached: db /= Void
	db_path_not_empty: not db_path.is_empty

end -- class SSC_DATABASE
